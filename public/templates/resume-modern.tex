% Name and address commands
\def\name#1{\def\@name{#1}}
\def\@name{}

\def\addressSep{$\diamond$}

\let\@addressone\relax
\let\@addresstwo\relax
\let\@addressthree\relax

\def\address#1{
  \@ifundefined{@addressone}{
    \def\@addressone{#1}
  }{
    \@ifundefined{@addresstwo}{
      \def\@addresstwo{#1}
    }{
      \def\@addressthree{#1}
    }
  }
}

\def\printaddress#1{
  \begingroup
    \def\\{\addressSep\ }
    \begin{center}#1\end{center}
  \endgroup
  \par
  \addressskip
}

\def\printname{
  \begingroup
    \begin{center}
      {\MakeUppercase{\namesize\bfseries \@name}}
    \end{center}
    \nameskip
  \endgroup
}

% Redefine document environment to print header
\let\ori@document=\document
\renewcommand{\document}{
  \ori@document
  \printname
  \@ifundefined{@addressone}{}{\printaddress{\@addressone}}
  \@ifundefined{@addresstwo}{}{\printaddress{\@addresstwo}}
  \@ifundefined{@addressthree}{}{\printaddress{\@addressthree}}
}

  }
  \item[]
}{
  \end{list}
}
